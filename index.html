document.querySelector('.portfolio-grid').addEventListener('click', function(e) { const expandBtn = e.target.closest('.project-expand'); if (!expandBtn) return; const projectCard = expandBtn.closest('.project-card'); const isExpanded = projectCard.getAttribute('aria-expanded') === 'true'; // Close all project cards document.querySelectorAll('.project-card').forEach(card => { card.setAttribute('aria-expanded', 'false'); }); // Toggle the clicked project card if it was not already expanded if (!isExpanded) { projectCard.setAttribute('aria-expanded', 'true'); } });
